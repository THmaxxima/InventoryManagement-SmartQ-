CREATE PROCEDURE proc_IVM_GetUnloadTruck_1862
       @MaterialTypeID Int,
       --@MaterialSourceID Int,
       @UserID Int
AS

BEGIN
       
       SET NOCOUNT ON;

       Select top 100 DI.DocID as Ticket,di.ReceiveDate,DITruck.TruckPlateLicense as PlateLicense
       ,di.MaterialID,mm.name as MaterialName
       ,disup.SupplierID,ms.name as SupplierName
       ,dqty.amount as Quantity,DMillWgt.Amount as MillWeight
       ,Case
              when di.DocStatusID In(4,7) then convert(bit,1)
              else convert(bit,0)
       end as IsWeightOut
       ,mSource.Name MaterialSource,UnloadStn.ReferenceValue as UnloadStation
       From DT_Inbound DI
       Join View_DTInbound_LatestVersionID V on V.ID = DI.ID 
       Join DT_InboundSupplier DISup on DISup.DT_InboundID=DI.ID AND DISup.SupplierTypeID=2
       Join DT_InboundTruck DITruck on DITruck.DT_InboundID=DI.ID 
       Join MT_Material mm on mm.id = di.MaterialID
       Join MT_Source mSource on mSource.ID = mm.SourceID 
       Join MT_Supplier ms on ms.id = DISup.SupplierID
       Left Join DT_InboundQuantity DQTY ON DQTY.DT_InboundID=DI.ID and DQTY.QuantityTypeID=4
       Left Join DT_InboundQuantity DMillWgt ON DMillWgt.DT_InboundID=DI.ID and DMillWgt.QuantityTypeID=2
       Left Join DT_InboundStorageBuffer DIBuff on DIBuff.Ticket=di.DocID
       Left Join DT_InboundDocument UnloadStn on UnloadStn.DT_InboundID =DI.ID  and UnloadStn.DocID =145
       where 1=1
       and mm.TypeID = @MaterialTypeID
       --and di.ReceiveDate > Dateadd(day,-5,Getdate())
       and di.DocStatusID in(1,4,7)
       and DIBUFF.ID IS NULL
       Order by di.id desc;
END
