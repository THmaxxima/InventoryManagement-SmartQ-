-- =============================================
-- Author:        Yuthapong Wongtharua
-- Create date: 29 Jan 2018
-- Description:    To do job move inventory from area to another area
/*
    Objective: create first for project 4.0 will call this proc to move stock to temp area and wait next day for confirm
    30 Apr 2018: Correct bug that it must see amount in buffer as source amount also.
    25 Jun 2018: Use StockDate in place of Insert date for movement
*/
-- =============================================
CREATE PROCEDURE proc_IVM_MoveStockTemp_1847
    @SourceStorageID Int,
    @DestinationStorageID Int,
    @MaterialID Int,
    @Amount float, -- just in case to use with other type than waste paper
    @UserID Int
AS

BEGIN

    SET NOCOUNT ON;
    Declare @JobName varchar(50)='MoveStockInField'
    Declare @StorageType_Area int=1;
    Declare @StorageType_Plant int=2;

    Declare @SourceTypeID Int;
    Declare @DestTypeID Int;
    Declare @UnitWeight float;
    Declare @MoveWeight float;
    Declare @StatusID_Create Int=1;
    Declare @MinStockDate Datetime;

    /* Weight will be used as average weight always
    */
    exec proc_WriteProcessLog @JobName,'Start',0,3,@UserID;
ValidationProcess:
    -- StorageType validation
    Select @SourceTypeID=m.MaterialTypeID
    From MT_StorageArea m
    where m.id=@SourceStorageID
    
    exec proc_WriteProcessLog @JobName,'Check Storage type',0,3,@UserID;
    If @SourceTypeID=@StorageType_Plant 
        Begin
            exec proc_WriteProcessLog @JobName,'Source storage cannot be plant',0,3,@UserID;
            RaisError('Source Storage cannot be Plant',11,11)
            Return
        End

    Select @DestTypeID=m.MaterialTypeID
    From MT_StorageArea m
    where m.id=@DestinationStorageID

    -- Source amount validation ... if it is enough to move
    exec proc_WriteProcessLog @JobName,'Source amount validation',0,3,@UserID;
    if not exists(Select 1 From DT_StorageAreaMaterial d
                where d.StorageID=@SourceStorageID and d.MaterialID=@MaterialID
                Group by d.StorageID,d.MaterialID
                Having sum(d.CurrentQty) >= @Amount)
        If not exists(Select 1 from DT_StorageAreaMaterialBuffer d
                     where d.DestinationStorageID =@SourceStorageID and d.MaterialID = @MaterialID)
            begin
                RaisError('Source Inventory is not enough to move',11,11)
                Return
            end
AggregateInventory:
    -- Actual inventory and buffer must be combined to get overall inventory to move
    Declare @DTInventory Table(ID Integer identity,StorageID Int,MaterialID Int,Bale float,[Weight] float,StockDate datetime)    
    Insert into @DTInventory (StorageID ,MaterialID ,Bale,[Weight],StockDate )
    Select  Inv.StorageID ,Inv.MaterialID ,SUM(Inv.Amount) as Amount,SUM(inv.[Weight]) as [Weight],MIN(Inv.MinStockDate) as StockDate
    From 
        (Select d.StorageID ,d.MaterialID,Sum(d.CurrentQty) as Amount,sum(d.CurrentWgt) as [Weight]
        ,MIN(Isnull(d.StockDate,d.InsertDate)) as MinStockDate
        From DT_StorageAreaMaterial d
        where d.MaterialID =@MaterialID 
        and d.StorageID = @SourceStorageID 
        Group by d.StorageID,d.MaterialID
        Union All -- add move in
        Select d.DestinationStorageID,d.MaterialID,Sum(d.Amount) as Amount,Sum(d.[Weight]) as [Weight]
        ,MIN(Isnull(d.StockDate,d.InsertDate)) as MinStockDate
        From DT_StorageAreaMaterialBuffer d
        where d.DestinationStorageID =@SourceStorageID 
        and d.MaterialID = @MaterialID 
        and d.StatusID =1
        Group by d.DestinationStorageID ,d.MaterialID 
        Union All -- deduct move out
        Select d.SourceStorageID,d.MaterialID,Sum(d.Amount)*(-1) as Amount,Sum(d.[Weight])*(-1) as [Weight]
        ,MIN(Isnull(d.StockDate,d.InsertDate)) as MinStockDate
        From DT_StorageAreaMaterialBuffer d
        where d.SourceStorageID =@SourceStorageID 
        and d.MaterialID = @MaterialID 
        and d.StatusID =1
        Group by d.SourceStorageID ,d.MaterialID ) Inv
    Group by Inv.StorageID,Inv.MaterialID 
    Having SUM(inv.[Weight]) >0 ;

    Declare @AvaiBale float
    Declare @TotalRec Int
    select @TotalRec = COUNT(d.id)
    From @DTInventory d;
    If @TotalRec =0 
            begin
                RaisError('Source Inventory is not enough to move',11,11)
                Return
            end
    else
        begin
            Select @AvaiBale = d.Bale ,@MinStockDate = d.StockDate 
            From @DTInventory d
            If @AvaiBale < @Amount 
                begin
                    RaisError('Source Inventory is not enough to move',11,11)
                    Return
                end
        end
    -- Find Average unit weight
    select @UnitWeight = d.[Weight]/d.Bale
    From @DTInventory d
    If @UnitWeight is null or @UnitWeight < 0
        begin
            RaisError('Unit Weight has problem. Please check unit count of source storage',11,11)
            Return
        end

    Set @MoveWeight = Round(@Amount * @UnitWeight,0);

MoveProcess:
    /*
        Selected storage will be saved at DT_StorageAreaMaterialbuffer with create status and wait for user to confirm in GUI
    */
    exec proc_WriteProcessLog @JobName,'Insert request to DT_StorageAreaMaterialBuffer',0,3,@UserID;
    Declare @BufferID Int
    exec X10_SP_GetNextID 'DT_StorageAreaMaterialBuffer',@UserID,@BufferID Output;
    Insert into DT_StorageAreaMaterialBuffer(ID,SourceStorageID,DestinationStorageID,MaterialID,Amount,[Weight]
                        ,StatusID,StockDate,InsertDate,UpDatedate,InsertUserID,UpdateUserID)
    Values(@BufferID,@SourceStorageID,@DestinationStorageID,@MaterialID,@Amount,@MoveWeight
            ,@StatusID_Create,@MinStockDate,Getdate() ,Getdate(),@UserID,@UserID);

    
    Print 'Success'

END
