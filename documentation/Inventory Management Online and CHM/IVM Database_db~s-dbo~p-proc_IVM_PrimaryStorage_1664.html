/*
    Retun as Group of Storage.
*/
CREATE procedure proc_IVM_PrimaryStorage_1664
    @FieldStorageID Int
as
    
Set NoCount on;

    Declare @DTRaw table(ID Int identity,GroupID Int,GroupName varchar(200),StorageCapacity float
    ,MaterialBalance float,MaterialWeight float
    ,InuseCapacity float,IsFull bit)

    -- Existing data
    Insert into @DTRaw (GroupID,GroupName,StorageCapacity
    ,MaterialBalance ,MaterialWeight
    ,InuseCapacity,IsFull)
    Select mg.id as GroupID,mg.Name as GroupName,Avg(mg.Capacity) as StorageCapacity
    ,Sum(DS.QTY) as MaterialBalance
    ,Round(Sum(ds.Wgt),3) as MaterialWeight
    ,Round(convert(float,Sum(ds.Wgt)*100)/convert(float,Avg(mg.Capacity)),1) as InUseCapacity
    ,Isnull(mg.FullCap,0) as IsFull
    From MT_StorageArea m
    Join MT_StorageAreaGroup mg on mg.ID = m.GroupID 
    Left Join
        (Select ds.StorageID,sum(ds.CurrentQty) Qty,sum(ds.CurrentWgt)/1000 Wgt
        From DT_StorageAreaMaterial ds
        Group by ds.StorageID) ds on ds.StorageID=m.ID
    where m.ParentStorageID In(0, @FieldStorageID)
    Group by mg.ID,mg.Name,mg.FullCap ;



    -- others group
    Insert into @DTRaw (GroupID,GroupName,StorageCapacity
    ,MaterialBalance ,MaterialWeight
    ,InuseCapacity,IsFull)
    Select 999999 as GroupID,'Other' as GroupName,Null as StorageCapacity
    ,Sum(DS.QTY) as MaterialBalance
    ,Round(Sum(ds.Wgt),3) as MaterialWeight
    ,Null as InUseCapacity
    ,CONVERT(bit,0) as IsFull
    From MT_StorageArea m
    Left Join
        (Select ds.StorageID,sum(ds.CurrentQty) Qty,sum(ds.CurrentWgt)/1000 Wgt
        From DT_StorageAreaMaterial ds
        Group by ds.StorageID) ds on ds.StorageID=m.ID
    where m.GroupID is null
    and m.ParentStorageID =@FieldStorageID ;




    -- Deduct/add more from buffer
    -- source will be deducted
    Declare @DTSource table(GroupID Int,TrQty float,TrWeight float)
    Insert into @DTSource(GroupID,TrQty,TrWeight)
    Select mg.id as GroupID,Sum(d.Amount) as Amount ,Sum(d.[Weight])/1000 as [Weight]
    From DT_StorageAreaMaterialBuffer d
    Join MT_Material mm on mm.ID = d.MaterialID 
    Join MT_StorageArea ms on ms.ID = d.SourceStorageID 
    Join MT_StorageAreaGroup mg on mg.ID = ms.GroupID 
    where d.StatusID =1
    Group by mg.ID ;
    
    -- add to destination
    -- destination will be added
    Declare @DTDest table(GroupID Int,TrQty float,TrWeight float)
    Insert into @DTDest(GroupID,TrQty,TrWeight)
    Select mg.id as GroupID,Sum(d.Amount) as Amount ,Sum(d.[Weight])/1000 as [Weight]
    From DT_StorageAreaMaterialBuffer d
    Join MT_Material mm on mm.ID = d.MaterialID 
    Join MT_StorageArea ms on ms.ID = d.DestinationStorageID
    Join MT_StorageAreaGroup mg on mg.ID = ms.GroupID 
    where d.StatusID =1
    and ms.ParentStorageID =@FieldStorageID
    Group by mg.ID ;
    -- add more from DT_InboundStorageBuffer
    Insert into @DTDest(GroupID,TrQty,TrWeight)
    Select mg.id as GroupID,Sum(d.Amount) as Amount ,Sum(d.[EstWeight]) as [Weight]
    From DT_InboundStorageBuffer d
    Join 
        (Select DI.DocID,DI.MaterialID 
        From DT_Inbound DI
        Join View_DTInbound_LatestVersionID V on V.ID = DI.ID ) DI on DI.DocID = d.Ticket
    Join MT_Material mm on mm.ID = di.MaterialID 
    Join MT_StorageArea ms on ms.ID = d.DestinationStorageID
    Join MT_StorageAreaGroup mg on mg.ID = ms.GroupID 
    where d.StatusID =1
    and ms.ParentStorageID =@FieldStorageID
    Group by mg.ID ;

CombineProcess:
    -- Deduct source
    Update d 
    Set d.MaterialBalance = d.MaterialBalance - Isnull(ds.TrQty,0) ,d.MaterialWeight = round(d.MaterialWeight - Isnull(ds.TrWeight,0) ,3)
    From @DTRaw d
    Join @DTSource ds on ds.GroupID = d.GroupID ;
    -- Add destination
    Update d 
    Set d.MaterialBalance = Isnull(d.MaterialBalance,0) + Isnull(ds.TrQty,0) 
    ,d.MaterialWeight = round(Isnull(d.MaterialWeight,0) + ds.TrWeight,3)
    From @DTRaw d
    Join @DTDest ds on ds.GroupID = d.GroupID ;
    -- New GroupID may be happened because user unload into new StorageID which is never exist before in invetory
    Insert into @DTRaw (GroupID,GroupName,StorageCapacity,MaterialBalance ,MaterialWeight,InuseCapacity,IsFull)
    Select d.GroupID,mg.Name as GroupName,mg.Capacity,d.TrQty,d.TrWeight,(d.TrWeight *100/Nullif(mg.Capacity ,0)) as InUseCapacity,mg.FullCap
    From @DTDest d
    Left Join @DTRaw dRaw on dRaw.GroupID = d.GroupID 
    Left join MT_StorageAreaGroup mg on mg.ID = d.GroupID 
    where dRaw.id is null;

    -- Check full cap
    Update d
    set d.InuseCapacity=round(d.MaterialWeight *100/NullIf(d.StorageCapacity,0),1)
    From @DTRaw d;

FinalStep:
    Select d.ID,d.GroupID ,d.GroupName ,d.StorageCapacity ,Isnull(d.MaterialBalance,0) as MaterialBalance 
    ,Isnull(d.MaterialWeight,0) MaterialWeight,Isnull(d.InuseCapacity,0) as InuseCapacity ,d.IsFull 
    from @DTRaw d
    order by d.GroupID;

    Set NoCount off;
