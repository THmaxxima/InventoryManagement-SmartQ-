/*
    Objective: In case user unload truck and change later because of something wrong. This can be done only or truck that was unloaded and not yet confirm in MPlan.
    25/6/2018: correct bug...  delete temp from DT_StorageAreaMaterialBuffer also
*/
CREATE PROCEDURE proc_IVM_RollBack_1894
    @ReferenceID Int,
    @RollBackTypeID Int, -- 1= Unload from truck, 2=From Movement
    @UserID Int
AS

BEGIN
    SET NOCOUNT ON;
    Declare @UserMsg varchar(500)
ValidationProcess:
    Declare @CurStatID Int=0

    If @RollBackTypeID =1
        begin
            Select @CurStatID=d.StatusID
            From DT_InboundStorageBuffer d
            where d.ID = @ReferenceID ;

            if @CurStatID=0
                begin
                    RaisError('Truck is not found',11,11)
                    Return
                end

            If @CurStatID<>1
                begin
                    RaisError('Unable to rollback. Unload process was confirmed',11,11)
                    Return
                end
            goto RollbackUnloadFromTruck
        end
    else
        begin
            goto RollbackFromMovement
        end
RollbackUnloadFromTruck:
    Declare @Ticket varchar(20);
    Declare @StockID Int;
    Select @StockID=d1.ID
    From DT_StorageAreaMaterialBuffer d1
    Join DT_InboundStorageBuffer d2 on d2.DestinationStorageID=d1.DestinationStorageID and d2.Amount=d1.Amount
    Join DT_Inbound di on di.DocID=d2.Ticket
    where d1.StatusID=1 and d1.MaterialID=di.MaterialID
    
    select @Ticket =d.Ticket 
    from DT_InboundStorageBuffer d
    where d.ID =@ReferenceID;

    Delete From DT_InboundStorageBuffer where ID=@ReferenceID;
    --- delete from DT_StorageAreaMaterialBuffer
    Delete From DT_StorageAreaMaterialBuffer  where ID=@StockID;


    Set @Usermsg ='User roll back ticket ' + @Ticket 
    exec proc_WriteProcessLog 'Delete Temp InboundStorage Unload',@UserMsg,0,1,@UserID
    goto FinalStep

RollbackFromMovement:
    Declare @SourceStorage varchar(20);
    Declare @DestStorage varchar(20);
    Declare @MaterialName varchar(50);
    

    select @SourceStorage =ms1.Name,@DestStorage =ms2.Name ,@MaterialName =mm.name
    from DT_StorageAreaMaterialBuffer d
    join MT_Material mm on mm.id = d.materialid
    join MT_StorageArea ms1 on ms1.ID = d.SourceStorageID 
    join MT_StorageArea ms2 on ms2.ID = d.SourceStorageID 
    where d.ID =@ReferenceID ;

    --Select * from mt_status;

    Delete from DT_StorageAreaMaterialBuffer where ID=@ReferenceID ;
    Set @Usermsg ='User roll back movement from storage ' + @SourceStorage + ' to ' + @DestStorage + ' material ' + @MaterialName 
    exec proc_WriteProcessLog 'Delete Temp InboundStorage movement',@UserMsg,0,1,@UserID
    goto FinalStep

FinalStep:
    --select * from DT_InboundStorageBuffer 
END
