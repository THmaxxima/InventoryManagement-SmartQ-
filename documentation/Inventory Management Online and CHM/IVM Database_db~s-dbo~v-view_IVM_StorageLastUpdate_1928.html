View Definition
create View view_IVM_StorageLastUpdate_1928
as


Select MAX(d.UpdateDate) as LatestUpdateDate
From
(
select MAX(d.updateDate) UpdateDate 
from DT_StorageAreaMaterialBuffer d

Union All

Select MAX(d.updateDate) UpdateDate 
from DT_StorageAreaMaterial d
Join MT_Material mm on mm.ID = d.MaterialID 
where mm.TypeID =1) d
